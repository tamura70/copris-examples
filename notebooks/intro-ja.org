* 始めての利用

#+BEGIN_SRC scala
import $cp.`copris-all-v2-3-1`
import jp.kobe_u.copris._
import dsl._

init
int('x, 1, 15)
int('y, 1, 15)
int('z, 1, 15)
add('x + 'y + 'z === 15)
add('x + 'y * 5 + 'z * 10 === 90)
#+END_SRC

#+BEGIN_SRC scala
println(csp)
println(csp.variables)
println(csp.bools)
println(csp.dom)
println(csp.constraints)
